create table milkyProductInfo(
    productCode number primary key,
    productName varchar(50),
    category varchar(30),
    productOption varchar(30),
    productUnit number,
    productRate number,
    enterCount number,
    sellCount number
);

create sequence milkyProductSeq nocache nocycle;


create table milkyProductDetail (
	productCode number,					-- 상품코드
	productEngName varchar(100),		-- 영어이름
	productSmallInfo varchar(200),		-- 상세정보1
	productDetailInfo varchar(500)		-- 상세정보2
);
